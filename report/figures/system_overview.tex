\documentclass{standalone}
\usepackage{tikz}

\begin{document}
\tikzstyle{startstop} = [rectangle, rounded corners, minimum width=3cm, minimum height=1cm,text centered, draw=black, fill=red!30]

\begin{tikzpicture}[node distance=2cm]

\node (Raw_EEG_Data) [io, below of=EEG_Monitor, yshift=0.5cm] {Raw EEG Data};

\node (Feature_Set) [io, below of=Processing, yshift=0.5cm] {Feature \\Set};

\draw [arrow] (HR_Monitor) -- (Raw_HR_Data);

\documentclass{standalone}
\usepackage{tikz}

\begin{document}

\tikzstyle{process} = [rectangle, minimum width=3cm, minimum height=1cm, text centered, text width=2cm, draw=black, fill=orange!30]

\begin{tikzpicture}[node distance=2cm]

\node (Partitioned_HR) [startstop] {Partitioned HR Data};

\node (Power) [io, below of=FFT] {Power Spectrum Data};

\draw [arrow] (Power) -- (EEG_Feature);

\draw [arrow] (Mean_HR) |- (Feature_Align);

\draw [arrow] (Channels) -- (Feature_Align);

\draw [arrow] (Feature_Align) -- (Features);
\end{tikzpicture}
